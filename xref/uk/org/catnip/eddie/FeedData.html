69      public String toString() {
70         StringBuilder ret = new StringBuilder();
71         //ret.append("bozo: " + error);
72         ret.append(property_map.toString());
73         if (author != null) {
74         ret.append("author = " + author + ", ");
75         }
76         if (info != null) {
77             ret.append("info = " + info + ", ");
78             }
79         if (image != null) {
80             ret.append("image = " + image + ", ");
81             }
82         if (textinput != null) {
83             ret.append("textinput = " + textinput + ", ");
84             }
85         if (generator != null) {
86             ret.append("generator = " + generator + ", ");
87         }
88         if (subtitle != null) {
89             ret.append("subtitle = " + subtitle + ", ");
90         }
91         if (!contributors.isEmpty()) {
92             ret.append("contributors = " + contributors + ", ");
93         }
94         if (!links.isEmpty()) {
95         ret.append("links = " + links + ", ");
96         }
97         if (getTitle() != null) {
98             ret.append("title = " + getTitle() + ", ");
99             }
100        if (!categories.isEmpty()){
101            ret.append("categories = " + categories + ", ");
102            }
103        if (!entries.isEmpty()) {
104        ret.append("entries = " + entries.toString());
105        }
106        
107        return ret.toString();
108     }
