99      public void addCategory(@NotNull Category category) {
100         Iterator<Category> it = categories.iterator();
101         while (it.hasNext()) {
102             Category cur_cat = (Category)it.next();
103             if (category.getTerm().equals(cur_cat.getTerm())) {
104                 return;
105             }
106         }
107         categories.add(category);
