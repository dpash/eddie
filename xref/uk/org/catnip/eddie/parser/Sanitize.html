93       * Class to parse html and remove unsafe elements and attributes
94       * @author David Pashley <david@davidpashley.com>

97      class HTMLSAXParser extends DefaultHandler2 {
98          private int errors = 0;
99          private boolean started_document = false;
100         private State state;
101 
102         private boolean unsafe_content = false;
103         private static final int MAX_ERRORS=2000;
104         @NotNull
105         private StringBuilder sb = new StringBuilder();
106         protected Locator locator;
107         public void setDocumentLocator(final Locator locator) {
108             this.locator = locator;
109         }
110         public void ignorableWhitespace(char[] ch, int start, int length) {
111             String data =  new String(ch, start,length);
112             log.trace("ignorableWhitespace: '"+ data+"'");
113             sb.append(data);
114         }
115         public void characters(char[] ch, int start, int length) {
116             if (unsafe_content) { return;}
117             String data =  new String(ch, start,length);
118             log.trace("characters: '"+ data+"'");
119             if (errors > 0 && data.contains(">")) {
120                 data = data.substring(data.indexOf(">")+1);
121             }
122             
123             sb.append(data);
124             
125         }
126         public void startElement(String uri, String localName, String qName,
127                 @NotNull Attributes atts) throws SAXException {
128             started_document = true;
129             // This is currently a hack. Should we find any more, 
130             // we'll need to create a hash
131             if ("image".equals(localName)) { localName = "img"; }
132             
133             log.trace("startElement: "+ localName);
134             if (!acceptable_elements.contains(localName)) { 
135                 if (unsafe_content_elements.contains(localName)) {
136                     unsafe_content = true;
137                 }
138                 return; 
139             }
140             sb.append("<");
141             sb.append(localName);
142             
143             
144             for (int i = 0; i < atts.getLength(); i++) {
145                 if (!acceptable_attributes.contains(atts.getLocalName(i))) {
146                     continue;
147                 }
148                 String attribute = atts.getLocalName(i);
149                 String value = atts.getValue(i);
150                 if (url_attributes.contains(attribute)) {
151                     value = state.resolveUri(Entities.decodeEntities(value));
152                 }
153                 sb.append(" ");
154                 sb.append(attribute);
155                 sb.append("=\"");
156                 sb.append(value);
157                 sb.append("\"");
158             }
159             if (elements_no_end_tag.contains(localName)) {
160                 sb.append(" /");
161             }
162             sb.append(">");
163         }
164         public void endElement(String uri, String localName, String qName)
165         throws SAXException {
166 
167             log.trace("endElement: "+ localName);
168             if (!acceptable_elements.contains(localName)) {
169                 if (unsafe_content_elements.contains(localName)) {
170                     unsafe_content = false;
171                 }
172                 return; 
173             }
174             if (elements_no_end_tag.contains(localName)) { return; }
175             sb.append("</");
176             sb.append(localName);
177             sb.append(">");
178         }
179         public void comment(char[] ch, int start, int length) throws SAXException {
180             if (!started_document) { return; }
181             String data =  new String(ch, start,length);
182             log.trace("comment: '"+ data+"'");            
183             sb.append("<!--"+data+"-->");
184         }
185         public void endDocument()
186         throws SAXException {
187             log.trace("endDocument()");
188 
189         }

210                 }

212         }

222     /**
223      * Clean unsafe elements and attributes from html. This is the main entrance to the 
224      * sanitizing subsystem
225      * @param data string to be cleaned
226      * @param state current state information
227      * @return cleaned string
228      */
229     static public String clean(String data, @NotNull State state) {
230         Sanitize s = new Sanitize();
231         data = data.replace("&", "&amp;");
232         if (state.getType().equals("text/plain") && (!"base64".equals(state.getMode()))) {
233             // Fix what we just broke
234             log.debug(state);
235             data = data.replace("&amp;lt;", "&lt;");
236             data = data.replace("&amp;gt;", "&gt;");
237         }
238         return s.real_clean(data, state);
239     }

242     /**
243      * This is the class that actually does the sanitizing
244      * @param data string to be cleaned
245      * @param state current state information
246      * @return cleaned string
247      */
248      public String real_clean(@NotNull String data, State state) {
249         String ret = data;
250         if (!data.trim().startsWith("<!DOCTYPE")) {
251             data = "<html>"+data+"</html>";
252         }
253         try {
254             
255             if (xr == null) {
256                 try {
257                     xr = (XMLReader)Class.forName("org.ccil.cowan.tagsoup.Parser").newInstance();
258                     xr.setFeature("http://www.ccil.org/~cowan/tagsoup/features/default-attributes", false);
259                 } catch (ClassNotFoundException ex) {
260                     log.warn("couldn't load tagsoup. Using standard sax parser " + ex);
261                     xr = new SAXParser();
262                     xr.setFeature("http://apache.org/xml/features/continue-after-fatal-error", true);
263                     xr.setFeature("http://apache.org/xml/features/nonvalidating/load-external-dtd",false);
264                 }
265             }
266             HTMLSAXParser handler = new HTMLSAXParser();
267             xr.setContentHandler(handler);
268             xr.setErrorHandler(handler);
269             xr.setProperty("http://xml.org/sax/properties/lexical-handler", handler);
270             
271             StringReader r = new StringReader(data);
272             handler.setState(state);
273             xr.parse(new InputSource(r));
274             ret = handler.getResult();
275         } catch (SAXParseException e) {
276             log.info("error", e);
277         } catch (SAXException e) {
278             log.debug("error", e);
279         } catch (Exception e) {
280             log.error("error", e);
281         }
282         
283         
284         log.debug("Cleaned: '"+data+"'");
285         log.debug("     to: '"+ret+"'");
286         return ret;
287     }

290     /**
291      * Rebuild text from a startElement SAX event
292      * Don't need to clean as this will happen later anyway.
293      * TODO: Should probably move this to BaseSAXParser
294      * @param state current sax event
295      * @return text representation of the SAX event
296      */
297     static public String clean_html_start(@NotNull State state) {
298         StringBuilder sb = new StringBuilder();    
299         sb.append("<");
300         sb.append(state.getElement());
301         for (int i = 0; i < state.getAttributes().getLength(); i++) {
302             String attribute = state.getAttributes().getQName(i);
303             String value = state.getAttributes().getValue(i);
304             sb.append(" ");
305             sb.append(attribute);
306             sb.append("=\"");
307             sb.append(value);
308             sb.append("\"");
309         }
310         if (elements_no_end_tag.contains(state.getElement())) {
311             sb.append(" /");
312         }
313         sb.append(">");
314         return sb.toString();
315     }
316 
317     /**
318      * Rebuild text from a endElement SAX event
319      * @param tag end tag
320      * @return text representation of the SAX event
321      */
322     @NotNull
323     static public String clean_html_end(String tag) {
324         if (elements_no_end_tag.contains(tag)) {
325             return "";
326         } else {
327             return "</" + tag + ">";
328         }
