81      /**
82       * Get the current state
83       * It returns a reference to the top item on the stack
84       * If the stack is empty (start of the document) we create a new 
85       * State object and set the base and language values so they are inherited
86       * throughout the document 
87       * @return the current state
88       */
89      @NotNull
90      protected State getCurrentState() {
91          if (current_state != null) { 
92              return current_state;
93          }
94          
95          if (!stack.empty()){
96              current_state = stack.peek();
97              return current_state;
98          } 
99          State current_state = new State();
100 
101         if (contentLocation != null){  
102             current_state.setBase(contentLocation);
103         } else {
104             current_state.setBase(filename);  
105         }
106         if(contentLanguage != null) {
107             current_state.setLanguage(contentLanguage);
108         }
109         return current_state;
110     }

113      * Add a State object onto the state stack
114      * If this is the first object on the stack set the base value
115      * from values we've been given by the caller

118     protected void push(final State state) {
119  
120         if (log.isTraceEnabled()) {
121             log.trace("pushing "+ state);
122         }
123         current_state = state;
124         stack.push(state);     
125     }

141     /** 
142      * Look for a method to deal with the state of the current element. 
143      * Creates a new state object.
144      * Deals with inline xml content
145      * If the element method fails to push the state, we do it for them. 
146      * We default to not expecting content
147      * @see org.xml.sax.ContentHandler#startElement(java.lang.String, java.lang.String, java.lang.String, org.xml.sax.Attributes)
148      */
149 	public void startElement(final String uri, final String localName, final String qName,
150            final Attributes atts) throws SAXException {
151         
152         State state = new State(uri, localName, qName, atts, getCurrentState());
153 
154         if (log.isTraceEnabled()) {
155             log.trace("startElement:" + localName + " (" + state.getElement() + ")");
156         }
157         
158         if (in_content > 0 && "escaped".equals(state.getMode())) {
159             state.setMode("xml");
160         }
161         if (in_content > 0 && "xml".equals(state.getMode())) {
162             handle_data(state, Sanitize.clean_html_start(state), false);
163             state.content = true;
164             push(state);
165             return;
166         }
167        
168         try {
169             Class[] argTypes = { State.class };
170             Object[] values = {state };
171             int stacksize = stack.size();
172             this.getClass().getMethod("startElement_" + state.getElement(), argTypes)
173                     .invoke(this, values);
174             if (stack.size() == stacksize) {
175                 //naughty handler didn't push state
176                 push(state);
177             }
178         } catch (NoSuchMethodException e) {
179             log.trace("unhandled element " + state.getElement());
180             state.setExpectingText(true);
181             push(state);
182             // } catch (InvocationTargetException e) {
183             // throw e.getCause();
184         } catch (Exception e) {
185             e.printStackTrace();
186         }
187     }

190     /** 
191      * Look for a method to deal with the end of the current element
192      * @see org.xml.sax.ContentHandler#endElement(java.lang.String, java.lang.String, java.lang.String)
193      */
194     public void endElement(final String uri, final String localName, final String qName)
195             throws SAXException {
196         
197         State state = new State(uri, localName, qName);
198         State prev = getCurrentState();
199         if(log.isTraceEnabled()){
200             log.trace("end_element: " + state);
201         }
202         
203         if (in_content > 0 && "escaped".equals(prev.getMode()) && prev.content) {
204             prev.setMode("xml");
205         }
206         if (in_content > 0 && "xml".equals(prev.getMode()) && prev.content) {
207             String data = pop(localName);
208             handle_data(prev, data
209                     + Sanitize.clean_html_end(state.getElement()), false);
210             return;
211         }
212          
213         try {
214             this.getClass().getMethod("endElement_" + state.getElement(), (Class[])null)
215                     .invoke(this, (Object[])null);
216         } catch (NoSuchMethodException e) {
217             log.trace("unhandled element " + state.getElement());
218             pop(state.getElement());
219         } catch (Exception e) {
220             e.printStackTrace();
221         }
222     }

224     /**
225      * Retrieve state off the state stack and fill information into the Detail object
226      * @param element name of the current element 
227      * @return the text content of the element
228      */
229     protected String pop(final String element) {
230         if (stack.empty()) { return ""; }
231    
232         if (!getCurrentState().getElement().equals(element)) { return "";}
233         State state = stack.pop();
234         current_state = null;
235         if (log.isTraceEnabled()) {
236             log.trace("popping " + state);
237         }
238         String output = state.getText(); 
239        
240         // TODO: we should give Detail the current stack item
241         // This would mean that we don't have BaseSAXParser knowing about 
242         // objects it shouldn't, making it less tied to RSS and Atom
243         detail = new Detail();
244         detail.setBase(state.getBase());
245         detail.setLanguage(state.getLanguage());
246         detail.setType(state.getType());
247         detail.setSrc(state.getAttr("src")); //TODO: this seems out of place
248         if (!output.equals("")){
249         detail.setValue(output);
250         }
251         if (!state.isExpectingText()) { return output; }
252         
253         if ("base64".equals(state.getMode())) {
254             output = new String(Base64.decodeBase64(output.trim().getBytes()));
255             detail.setValue(output);
256         }
257         if ("application/octet-stream".equals(state.getType())) {
258             output = new String(Base64.decodeBase64(output.trim().getBytes()));
259             detail.setValue(output);
260         }
261         
262         if (in_content > 0) {
263             output = Sanitize.clean(output, state);
264             detail.setValue(output);
265         }
266         return output;
267         
268     }
269     /** 
270      * Deal with non markup characters
271      * @see org.xml.sax.ContentHandler#characters(char[], int, int)
272      */
273     public void characters(char[] ch, int start, int length) {
274         String data = new String(ch, start, length);
275 
276         if (in_content > 0) {
277             if (getCurrentState().getType().equals("application/xhtml+xml")) {
278                 if (data.equals("<")) { data = "&lt;"; }
279                 if (data.equals(">")) { data = "&gt;"; }
280                 if (data.equals("&")) { data = "&amp;"; }
281             } else if (getCurrentState().getType().equals("text/plain")) {
282                 if (data.equals("<")) { data = "&lt;"; }
283             }
284         }
285 
286         if(log.isTraceEnabled()) {
287             log.trace("characters: " + data);
288         }
289         getCurrentState().addText(data);
290     }

293     public void handle_data(@NotNull State state, String data, boolean escape) {
294         if (stack.empty()) { return; }
295         if (escape && state.getType().equals("application/xhtml+xml")) {
296            data = xmlescape(data);
297         }
298         
299         getCurrentState().addText(data);
300     }
301     
302     @NotNull
303     public String xmlescape(@NotNull String data) {
304         data.replace("&", "&amp;");
305         data.replace("<", "&lt;");
306         data.replace(">", "&gt;");
307         return data;
308     }
