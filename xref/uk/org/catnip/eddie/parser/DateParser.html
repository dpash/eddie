77  	@NotNull
78      static String[] date_formats = { "yyyy-MM-dd'T'kk:mm:ss'Z'", // ISO
79  			"yyyy-MM-dd'T'kk:mm:ssz", // ISO
80  			"yyyy-MM-dd'T'kk:mm:ss", // ISO
81  			"EEE, d MMM yy kk:mm:ss z", // RFC822
82  			"EEE, d MMM yyyy kk:mm:ss z", // RFC2882
83  			"EEE MMM  d kk:mm:ss zzz yyyy", // ASC
84  			"EEE, dd MMMM yy kk:mm:ss",

87  			"yyyy-MM-dd kk:mm:ss.0", "-yy-MM", "-yyMM", "yy-MM-dd",
88  			"yyyy-MM-dd", "yyyy-MM", "yyyy-D", "-yyMM", "yyyyMMdd", "yyMMdd",
89  			"yyyy", "yyD"

93  	@Nullable
94      public static Date parse(String d) {
95  		Date date = null;
96  		SimpleDateFormat formatter = new SimpleDateFormat();
97  		formatter.setTimeZone(TimeZone.getTimeZone("GMT"));
98  		d = d.replaceAll("([-+]\\d\\d:\\d\\d)", "GMT$1"); // Correct W3C times
99  		d = d.replaceAll(" ([ACEMP])T$", " $1ST"); // Correct Disney times
100         for (String date_format : date_formats) {
101             try {
102                 if (log.isTraceEnabled()) {
103                     log.debug("trying '" + date_format + "'");
104                 }
105                 formatter.applyPattern(date_format);
106                 date = formatter.parse(d);
107                 break;
108             } catch (Exception e) {
109                 if (log.isTraceEnabled()) {
110                     log.debug("parsing of date failed");
111                 }
112                 // Oh well. We tried
113             }
114         }
115 
116 		return date;
