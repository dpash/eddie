192     public void endElement_entry() throws SAXException {
193         pop("entry");
194 
195         if (current_entry.get("summary") == null && current_entry.get("description") != null){
196             current_entry.set("summary", current_entry.get("description"));
197         } else if (current_entry.get("summary") != null && current_entry.get("description") == null){
198             current_entry.set("description", current_entry.get("summary"));
199         }
200         if (current_entry.get("id") == null && current_entry.hasEnclosures()) {
201             Enclosure enclosure = (Enclosure)current_entry.getEnclosures().get(0);
202             current_entry.set("id", enclosure.getUrl());
203         }
204         if (current_entry.getAuthor() == null) {
205         	current_entry.setAuthor(feed.getAuthor());
206         }
207         feed.addEntry(current_entry);
208         current_entry = null;
209     }

309     public void endElement_link() throws SAXException {
310         State state = getCurrentState();
311         String content = pop("link");
312         content = state.resolveUri(content);
313         if (in_image) {
314             image.setLink(content);
315         } else if (textinput != null) {
316             textinput.setLink(content);
317         } else if (link != null){ // TODO cleanup
318             if (current_entry != null) {
319                 current_entry.setGuidIsLink(false);
320             }
321             link.setDetails(detail);
322             getCurrentContext().addLink(link);
323             if (link.getHref() != null) {
324 
325                 if ((link.getType().equals("text/html") || link.getType()
326                         .equals("application/xhtml+xml"))
327                         && ("alternate".equals(link.getRel()))) {
328 
329                     getCurrentContext().set("link", link.getHref());
330                 }
331                 if ("alternate".equals(link.getRel())) {
332                     getCurrentContext().set("link", link.getHref());
333                     
334                 }
335             } else {
336                 getCurrentContext().set("link", content);
337             }
338             link = null;
339         }
340         
341 
342     }

461     public void startElement_channel(@NotNull State state) throws SAXException {
462         in_feed = true;
463         if (null != state.getAttr("base")) {
464             state.setBase(state.getAttr("base"));
465         }
466         if (null != state.getAttr("lastmod")) {
467             feed.set("date",state.getAttr("lastmod"));
468             feed.set("modified",state.getAttr("lastmod"));
469             feed.setModified(DateParser.parse(state.getAttr("lastmod")));
470         }
471         if (null != state.getAttr("rdf:about")) {
472             feed.set("uri",state.resolveUri(state.getAttr("rdf:about")));
473             link = new Link();
474             link.setHref(state.resolveUri(state.getAttr("rdf:about")));
475             feed.addLink(link);
476             link = null;
477         }
478         if (null != state.getAttr("href")) {
479             feed.set("link",state.resolveUri(state.getAttr("href")));
480             link = new Link();
481             link.setHref(state.resolveUri(state.getAttr("href")));
482             feed.addLink(link);
483             link = null;
484         }
485 
486     }

573     public void startElement_feed(@NotNull State state) throws SAXException {
574 
575         this.in_feed = true;
576         String attr_version = state.getAttr("version");
577         
578         if (!this.feed.has("format"))
579             if (attr_version == null || attr_version.equals("")) {
580                 if ("http://www.w3.org/2005/Atom".equals(state.getUri())) {
581                     this.feed.set("format", "atom10");
582                 } else {
583                     this.feed.set("format", "atom");
584                 }
585             } else {
586                 if (attr_version.equals("0.1")) {
587                     this.feed.set("format", "atom01");
588                 } else if (attr_version.equals("0.2")) {
589                     this.feed.set("format", "atom02");
590                 } else if (attr_version.equals("0.3")) {
591                     this.feed.set("format", "atom03");
592                 } else {
593                     this.feed.set("format", "atom");
594                 }
595             }
596         if (state.getLanguage() != null) {
597             this.feed.set("language", state.getLanguage());
598         }
599         
600         // Do some sanity checking
601         assert(this.feed.has("format")) : "Failed to detect Atom format";
602         
603     }

666     public void startElement_link(@NotNull State state) throws SAXException {
667         if ("enclosure".equals(state.getAttr("rel"))) {
668             Enclosure enclosure = new Enclosure();
669             enclosure.setUrl(state.resolveUri(state.getAttr("href")));
670             enclosure.setLength(state.getAttr("length"));
671             enclosure.setType(state.getAttr("type"));
672             if (current_entry != null) {
673                 current_entry.addEnclosure(enclosure);
674             }
675         } else if ("image".equals(state.getAttr("rel"))) {
676             Image image = new Image();
677             image.setUrl(state.getAttr("href"));
678             getCurrentContext().setImage(image);
679         } else if ("self".equals(state.getAttr("rel"))) {
680         	this.getCurrentState().setBase(state.resolveUri(state.getAttr("href")));
681         }
682         link = new Link();
683         link.setHref(state.resolveUri(state.getAttr("href")));
