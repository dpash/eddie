297     @Nullable
298     public String detect(byte[] data) {
299         String encoding = null;
300         
301         // check for a <? as the first two byte
302         if (data[0] == 0x3c && data[1] == 0x3f) { 
303             encoding = detectUsingXML(data);
304         } else {
305             encoding = detectUsingMagic(data);
306             if ("EBCDIC".equals(encoding)) {
307                 encoding = detectEBCDIC(data);
308             }
309         }
310         encoding = alias(encoding);
311         if (encoding == null) {
312             encoding = defaultEncoding;
313         }
314         try {
315         	@SuppressWarnings("unused")
316 			Charset charset = Charset.forName(encoding);
317         } catch (UnsupportedCharsetException ex) {
318         	log.info("bogus encoding: " + encoding +". Using default");
319         	encoding = defaultEncoding;
320         	error = true;
321         }
322 
323         log.debug("detected: '" + encoding +"'");
324         return encoding;
325     }

332     public String alias(final String encoding) {
333         String new_encoding = encoding;
334         if (encoding_map.containsKey(encoding)) {
335             new_encoding = encoding_map.get(encoding);
336         }
337         return new_encoding;

340     @Nullable
341     private String detectUsingXML(byte[] data) {
342         String encoding = null;
343         try {
344 
345             BufferedReader reader = new BufferedReader(new InputStreamReader(
346                     new ByteArrayInputStream(data)));
347 
348             String str = reader.readLine();
349 
350             Pattern pattern = Pattern.compile(".*encoding=\"([^\"]*)\".*");
351             Matcher matcher = pattern.matcher(str);
352             if (matcher.matches()) {
353                 encoding = matcher.group(1);
354             }
355         } catch (IOException e) {
356             e.printStackTrace();
357         }
358         return encoding;
359     }

361     @Nullable
362     private String detectUsingMagic(byte[] data) {
363         String encoding = null;
364         int index = 0;
365         for (Magic magic : magics) {
366             if (checkArray(data, magic.magic)) {
367                 encoding = magic.name;
368                 if(magic.hasBOM) {
369                     bomLength = magic.magic.length;
370                 }
371                 break;
372             }
373             index++;
374         }
375         return encoding;
376     }

379         boolean equals = true;

382                 equals = false;

385         }
