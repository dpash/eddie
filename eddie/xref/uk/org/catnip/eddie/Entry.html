59      public String toString() {
60          StringBuilder ret = new StringBuilder();
61          ret.append("{\n");
62          Iterator<String> it = property_map.keySet().iterator();
63          while (it.hasNext()){
64              String key = (String)it.next();
65              String value = property_map.get(key);
66              ret.append("\t");
67              ret.append(key);
68              ret.append(" = '");
69              if (value.length() > 100) {
70                  ret.append(value.substring(0,100));
71              } else {
72                  ret.append(value);
73              }
74              ret.append("',\n");
75          }
76          // ret.append(property_map.toString());
77  
78          if (!content.isEmpty()) {
79              ret.append("\tcontent =");
80              ret.append(content);
81              ret.append(",\n");
82          }
83          if (this.getTitle() != null) {
84              ret.append("\ttitle_detail = '");
85              ret.append(this.getTitle());
86              ret.append("',\n");
87          }
88          if (this.getSummary() != null) {
89              ret.append("\tsummary_detail = '");
90              ret.append(this.getSummary());
91              ret.append("',\n");
92          }
93          if (!contributors.isEmpty()) {
94              ret.append("\tcontributors = ");
95              ret.append(contributors);
96              ret.append(",\n");
97          }
98          if (!links.isEmpty()) {
99              ret.append("\tlinks = ");
100             ret.append(links);
101             ret.append("',\n");
102         }
103         if (!categories.isEmpty()) {
104             ret.append("\tcategories = ");
105             ret.append(categories);
106             ret.append("',\n");
107         }
108         if (!enclosures.isEmpty()) {
109             ret.append("\tenclosures = ");
110             ret.append(enclosures);
111             ret.append("',\n");
112         }
113         if (getCreated() != null) {
114             ret.append("\tcreated = '");
115             ret.append(getCreated());
116             ret.append("'\n");
117         }
118         if (getIssued() != null) {
119             ret.append("\tissued = '");
120             ret.append(getIssued());
121             ret.append("'\n");
122         }
123         if (getExpired() != null) {
124             ret.append("\texpired = '");
125             ret.append(getExpired());
126             ret.append("'\n");
127         }
128         if (getModified() != null) {
129             ret.append("\tmodifed = '");
130             ret.append(getModified());
131             ret.append("'\n");
132         }
133         ret.append("}");
134         return ret.toString();
135     }
