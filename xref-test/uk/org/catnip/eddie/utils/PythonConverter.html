116     public static boolean runPython(@NotNull FeedData feed, String test) throws Exception {
117 
118         if (feed.error) {
119             interp.set("bozo", new PyInteger(1));
120         } else {
121             interp.set("bozo", new PyInteger(0));
122         }
123 
124         PyList entries_list = new PyList();
125         List<Entry> entries = feed.getEntries();
126         for (Entry entry: entries) {
127             entries_list.append(convertEntry(entry));
128         }
129         if (feed.get("format") != null) {
130         interp.set("version", new PyString(feed.get("format")));
131         }
132         if (feed.get("encoding") != null) {
133             interp.set("encoding", new PyString(feed.get("encoding")));
134             }
135         interp.set("feed", convertFeed(feed));
136         interp.set("entries", entries_list);
137         interp.exec("ret ="+ test);
138         PyInteger ret = (PyInteger)interp.get("ret");
139 		return ret.getValue() != 0;

144         PyDictionary feed_dict = new PythonConverter().new EddieDict();
145         Iterator<String> it = feed.keys();
146         while (it.hasNext()) {
147             String key = (String) it.next();

155         if (feed.getAuthor() != null) {
156             feed_dict.__setitem__("author_detail", convertAuthor(feed.getAuthor()));
157         }
158         if (feed.getPublisher() != null) {

167         }

169             feed_dict.__setitem__("info_detail",convertDetail(feed.getInfo()));
170         }
171         if (feed.getGenerator() != null) {
172             feed_dict.__setitem__("generator_detail",convertGenerator(feed.getGenerator()));
173         }

175             feed_dict.__setitem__("copyright_detail",convertDetail(feed.getCopyright()));

177         }

179             feed_dict.__setitem__("created_parsed",convertDate(feed.getCreated()));
180         }

185             feed_dict.__setitem__("modified_parsed",convertDate(feed.getModified()));

189             feed_dict.__setitem__("image",convertImage(feed.getImage()));
190         }
191         if (feed.getTextinput() != null) {
192             feed_dict.__setitem__("textinput",convertTextInput(feed.getTextinput()));
193         }

195             feed_dict.__setitem__("cloud",convertCloud(feed.getCloud()));

207         }

212             category_list.append(convertCategoryTuple(category));
213         }

218         for (Category category: feed.getCategories()) {
219             category_list.append(convertCategory(category));
220         }
221         feed_dict.__setitem__("tags",category_list);
222         
223         log.debug(feed_dict);
224

626     static public int normaliseDayOfWeek(int day) {
627         if (day == Calendar.MONDAY) { return 0; }
628         if (day == Calendar.TUESDAY) { return 1; }
629         if (day == Calendar.WEDNESDAY) { return 2; }
630         if (day == Calendar.THURSDAY) { return 3; }
631         if (day == Calendar.FRIDAY) { return 4; }
632         if (day == Calendar.SATURDAY) { return 5; }
633         if (day == Calendar.SUNDAY) { return 6; }
634         return 0;
635     }
