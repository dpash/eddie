66  	@NotNull
67      @Parameterized.Parameters
68  	public static Collection<String[]> findTests() throws IOException {
69  		List<String[]> files = new LinkedList<String[]>();
70  		for (Enumeration<URL> e = EddieTest.class.getClassLoader().getResources(
71  				test_dir); e.hasMoreElements();) {
72  			URL url = e.nextElement();
73  			List<File> fl = getFileList(new File(url.getFile()));
74  			for (File f : fl) {
75  				String file = null;
76  				try {
77  					file = url.toURI().relativize(f.toURI()).toString();
78  				} catch (URISyntaxException e1) {
79  					e1.printStackTrace();
80  				}
81  				files.add(new String[]{file, test_dir + "/" + file});
82  			}
83  
84  		}
85  		return files;
86  	}

89  	@NotNull
90      protected static List<File> getFileList(@NotNull File file) {
91  		List<File> files = new LinkedList<File>();
92  		if (file.isDirectory()) {
93  			File filelist[] = file.listFiles();
94  			for (File f : filelist) {
95  				files.addAll(getFileList(f));
96  			}
97  		} else {
98  			files.add(file);
99  		}
100 		return files;
101 	}

121 			Matcher test_matcher = test_pattern.matcher(line);

125 			} else if (header_matcher.matches()) {
